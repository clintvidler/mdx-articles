```go
byteData, err := io.ReadAll(r.Body)
if err != nil {
  fmt.Printf("Byte parse error: %v", err)
  return
}

var jsonData bytes.Buffer
if err := json.Indent(&jsonData, byteData, "", "\t"); err != nil {
  fmt.Printf("Json parse error: %v", err)
  return
}

fmt.Println(jsonData.String())
```
