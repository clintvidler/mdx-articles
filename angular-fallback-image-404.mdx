Try to load an image, if it doesn't exist, load fallback image.

`.html`

```html
<img
  [src]="getImage(v)"
  alt="Photo of {{ v.name }}"
  (error)="errImage($event, v)"
/>
```

`.ts`

```ts
errImage(e: any, v: any): void {
  e.target.src = this.getAltImage(v);
}
```
